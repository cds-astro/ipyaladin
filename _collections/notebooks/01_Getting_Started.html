Getting Started#

Displays an interactive sky view of the DECam Plane Survey.

[1]:

from ipyaladin import Aladin

[2]:

aladin = Aladin(
    survey="CDS/P/DECaPS/DR1/color",
    show_coo_grid=True,
    target="galactic center",
    coo_frame="galactic",
    fov=40,
    height=600,
)
aladin
